CREATE TABLE fact_order_items AS
SELECT
    oi.id AS order_item_id,
    oi.order_id,
    o.user_id,
    oi.product_id,
    p.category_id,
    oi.quantity,
    oi.price,
    oi.quantity * oi.price AS total_amount,
    o.created_at AS order_date
FROM order_items oi
JOIN orders o ON oi.order_id = o.id
JOIN products p ON oi.product_id = p.id;


CREATE TABLE dim_users AS
SELECT
    id AS user_id,
    name,
    email,
    created_at AS signup_date
FROM users;


CREATE TABLE dim_products AS
SELECT
    id AS product_id,
    name,
    price,
    category_id
FROM products;


CREATE TABLE dim_categories AS
SELECT
    id AS category_id,
    name AS category_name
FROM categories;


CREATE INDEX idx_fact_order_user ON fact_order_items(user_id);
CREATE INDEX idx_fact_order_product ON fact_order_items(product_id);
CREATE INDEX idx_fact_order_category ON fact_order_items(category_id);


SELECT
    c.category_name,
    SUM(f.total_amount) AS total_sales
FROM fact_order_items f
JOIN dim_categories c ON f.category_id = c.category_id
GROUP BY c.category_name
ORDER BY total_sales DESC;


SELECT
    u.name,
    SUM(f.total_amount) AS total_spent
FROM fact_order_items f
JOIN dim_users u ON f.user_id = u.user_id
GROUP BY u.name
ORDER BY total_spent DESC
LIMIT 10;

SELECT
    p.name,
    SUM(f.quantity) AS total_quantity
FROM fact_order_items f
JOIN dim_products p ON f.product_id = p.product_id
GROUP BY p.name
ORDER BY total_quantity DESC
LIMIT 10;
