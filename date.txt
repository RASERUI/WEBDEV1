SELECT DATE_TRUNC('month', f.order_date) AS month, SUM(f.total_amount) AS revenue
FROM fact_order_items f
GROUP BY month
ORDER BY month;

SELECT c.category_name, SUM(f.total_amount) AS total_spent
FROM fact_order_items f
JOIN dim_categories c ON f.category_id = c.category_id
WHERE f.user_id = 1
GROUP BY c.category_name
ORDER BY total_spent DESC;
